function submitQuiz() { let skor = 0; soalTerpilih.forEach((item, index) => { const pilihan = document.querySelector(`input[name="soal${index}"]:checked`); const feedback = document.getElementById(`feedback${index}`); if (pilihan && pilihan.value === item.jawaban) { skor++; feedback.textContent = "Benar"; feedback.classList.remove("text-red-600"); feedback.classList.add("text-green-700"); } else { feedback.textContent = "Salah"; feedback.classList.remove("text-green-700"); feedback.classList.add("text-red-600"); } }); const nilai = Math.round((skor / soalTerpilih.length) * 100); document.getElementById("score").innerText = `Kamu menjawab benar ${skor} dari ${soalTerpilih.length} soal. Nilai kamu: ${nilai}`; }
